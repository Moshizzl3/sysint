CREATE VIEW tranction_view AS (
 SELECT u.user_name,
    a.account_number,
    t.amount
   FROM ((users u
     JOIN accounts a ON ((a.user_id = u.id)))
     JOIN transactions t ON ((t.account_id = a.id)))
)
