```sql
CREATE VIEW character_details_view AS (
 SELECT c.id AS character_id,
    c.character_name,
    c.gender,
    bc.name AS class_name,
    get_character_place(c.id) AS place_name,
    c.alive,
    c.level,
    c.xp,
    c.money
   FROM ("character" c
     JOIN base_class bc ON ((c.base_class_id = bc.id)))
)
```
